def initialize(lists = nil)
  if !lists.nil?
    @lists = lists
  end
  @file_type = 'CSV'
  @sort_by = 'EMAIL_ADDRESS'
  @export_date_added = true
  @export_added_by = true
  @column_names = ['Email Address', 'First Name', 'Last Name']
end
